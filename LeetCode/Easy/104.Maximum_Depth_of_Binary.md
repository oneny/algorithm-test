## 내가 푼 솔루션

```js
var maxDepth = function(root) {
    if (!root) return 0;
    let depth = 0;
    const queue = [root];

    while (queue.length) {
        let len = queue.length;

        while (len--) { // 같은 뎁스에 대해서 일단 다 넣어주네
            const curr = queue.shift();
            if (curr.left) queue.push(curr.left);
            if (curr.right) queue.push(curr.right);
        }
        depth++;
    }

    return depth;
};
```
